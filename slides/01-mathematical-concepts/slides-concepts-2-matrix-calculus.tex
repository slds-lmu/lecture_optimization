\begin{framev}{Scope}
\begin{itemize}
\setlength{\itemsep}{0.5\baselineskip}
\item $\Xspace$/$\Yspace$ denote space of \textbf{independent}/\textbf{dependent} variables
\item Identify dependent variable $y$ with a \textbf{function} $f: \Xspace \to \Yspace, x\mapsto f(x)$
\item Assume $y$ sufficiently smooth
\item In matrix calculus, $x$ and $y$ can be \textbf{scalars}, \textbf{vectors}, or \textbf{matrices}
\item We denote vectors/matrices in \textbf{bold} lowercase/uppercase letters
\spacer
\begin{table}
\centering
\begin{tabular}{c||c|c|c}
Type & scalar $x$ & vector $\xv$ & matrix $\mathbf{X}$ \\ \hline\hline
scalar $y$ & $d y / d x$ & $d y / d\xv$ & $d y / d\mathbf{X}$ \\ \hline
vector $\yv$ & $d\yv / d x$ & $d\yv / d\xv$ & -- \\ \hline
matrix $\mathbf{Y}$ & $d\mathbf{Y} / d x$ & -- & --
\end{tabular}
\end{table}
\item This notation is also referred to as \emph{Leibniz notation}
\end{itemize}
\end{framev}


\begin{framev}{Leibniz Notation convention}
\begin{itemize} \setlength{\itemsep}{0.5\baselineskip}
\item Instead of writing $f(x)$ everywhere, we replace the function $f$ with the variable $y$.
\item This helps clarify relationships when multiple functions or variables are involved, especially in contexts like partial derivatives or matrix calculus.
\item Also applicable to partial derivatives: For $y = f:\R^n\longrightarrow\R,\; \xv\mapsto\fx=f(x_1, x_2, \ldots, x_n)$, the partial derivative of $y$ w.r.t. $x_i$ is $d y/d x_i$.
\item \textbf{Examples:}
\begin{itemize}
\item    $
y = x^3 + 5x \Longrightarrow
\dfrac{dy}{dx} = 3x^2 + 5
$
\spacer
\item $y = x_1^2 + 3x_2 \Longrightarrow
\dfrac{d y}{d x_1} = 2x_1, \quad
\dfrac{d y}{d x_2} = 3$
\end{itemize}
\end{itemize}
\end{framev}

\end{tabular}

\begin{itemize}
\item $d y / d\xv$ is the gradient from the previous slide deck
\spacer
\item When the input is a matrix the concept remains the same, i.e. for $y= f:\R^{m\times n}\longrightarrow\R,\; \mathbf{X}\mapsto f(\mathbf{X})$
$$
\frac{d y}{d\bm{X}}= \left(\begin{array}{ccc}
\dfrac{\partial f}{\partial x_{11}} & \cdots & \dfrac{\partial f}{\partial x_{1 n}} \\
\vdots & \ddots & \vdots \\
\dfrac{\partial f}{\partial x_{m 1}} & \cdots & \dfrac{\partial f }{\partial x_{m n}}
\end{array}\right) \in \R^{m \times n}
$$
\end{itemize}
\end{framev}

\begin{framev}{Deriv. of functions with scalar inputs}

matrix $\mathbf{Y}$ & \cellcolor{blue!20}$d\mathbf{Y} / d x$ & -- & --
\end{tabular}
\end{table}
\begin{itemize}
\item  Here, for univariate $f_{ij}:\R\rightarrow\R$, $\yv$ ($n=1$) or $\mathbf{Y}$ ($n>1$) are equal to a function $f: \R\longrightarrow\R^{m\times n}, x\mapsto \big(f_{ij}(x)\big)_{i=1,\dots,m;\,j=1,\dots,n}$ and the derivatives are, respectively given by $$
\frac{d\yv}{d x}=\begin{pmatrix}
\dfrac{\partial f_{1}}{\partial x}\\\vdots\\ \dfrac{\partial f_{m}}{\partial x}
\end{pmatrix}\in\R^m;\quad
\frac{d\mathbf{Y}}{d x}=\left(\begin{array}{ccc}
\dfrac{\partial f_{11}}{\partial x} & \cdots & \dfrac{\partial f_{1 n}}{\partial x} \\
\vdots & \ddots & \vdots \\
\dfrac{\partial f_{m 1}}{\partial x} & \cdots & \dfrac{\partial f_{m n}}{\partial x}
\end{array}\right) \in \R^{m \times n}$$
\end{itemize}
\end{framev}


\begin{framev}{Multivariate Differentiation Rules}
\begin{itemize}
\item Basic rules from single-variable calculus still apply.
\item But, for $\xv\in\R^{n}$: gradients are vectors/matrices (order matters).
\end{itemize}
\spacer
\textbf{Key Rules:}\\
\begin{itemize}
\item \textbf{Sum:}
$$
\frac{d}{d \xv}\bigl(f+g\bigr)=\frac{d f}{d \xv}+\frac{d g}{d \xv}
$$
\item \textbf{Product:}
$$
\frac{d}{d \xv}\bigl(fg\bigr)=\frac{d f}{d \xv}\,g+f\,\frac{d g}{d \xv}
$$
\item \textbf{Chain:}
$$
\frac{d}{d \xv}\Bigl((f\circ g)(\xv)\Bigr)=\dfrac{d}{d \xv}(f(g(\xv)))=\frac{d f}{d g}\,\frac{d g}{d \xv}
$$
\end{itemize}
\end{framev}


\begin{framei}[fs=small]{Details on the Chain Rule}
\item Suppose\begin{itemize}
\item  we have functions $\boldsymbol{g}: S\subseteq\R^n \rightarrow \R^m$ and $\boldsymbol{f}: T\subseteq\R^m \rightarrow \R^{\ell}$
\item $\boldsymbol{a} \in S$ is a point such that $\boldsymbol{g}(\boldsymbol{a}) \in T$ $\Rightarrow$ $\boldsymbol{f} \circ \boldsymbol{g}(\xv)=\boldsymbol{f}(\boldsymbol{g}(\xv))$ is well-defined for all $\xv$ close to $\boldsymbol{a}$
\end{itemize}
\item Then, if $\boldsymbol{g}$ is differentiable at $\boldsymbol{a}$ and $\boldsymbol{f}$ is differentiable at $\boldsymbol{g}(\boldsymbol{a})$\\
$\Rightarrow\boldsymbol{f} \circ \boldsymbol{g}$ is differentiable at $\boldsymbol{a}$, and the derivative $\dfrac{d \boldsymbol f}{d \boldsymbol g} \dfrac{d \boldsymbol g}{d \xv}$, is equal to
\begin{align*}
\nabla_{\boldsymbol{a}} \boldsymbol{f} \circ \boldsymbol{g}\;\hat{=}\;\boldsymbol{J}_{\boldsymbol{f} \circ \boldsymbol{g}}(\boldsymbol{a})=\boldsymbol{J}_{\boldsymbol{f}}(\boldsymbol{g}(\boldsymbol{a}))\boldsymbol{J}_ {\boldsymbol{g}}(\boldsymbol{a})\;\hat{=}\;\nabla_{\boldsymbol{g}(\boldsymbol{a})}{\boldsymbol{f}}\;\nabla_{\boldsymbol{a}} {\boldsymbol{g}}\in\R^{\ell\times n}
\end{align*}
\begin{footnotesize}
(See Chapter 2.3 of the UofT course \emph{MAT237 - Multivariable Calculus} for proof.) %
\end{footnotesize}
\item We can also write $\boldsymbol{f}$ as a function of $\yv=\left(y_1, \ldots, y_m\right) \in \R^m$, and $\boldsymbol{g}$ as a function of $\xv=\left(x_1, \ldots, x_n\right) \in \R^n$. Then for each $k=1, \ldots, \ell$ and $j=1, \ldots, n$
$$
\left[\boldsymbol{J}_{\boldsymbol{f} \circ \boldsymbol{g}}(\boldsymbol{a})\right]_{kj} = \frac{\partial}{\partial x_j}\left(f_k \circ \boldsymbol{g}\right)(\boldsymbol{a})=\sum_{i=1}^m \frac{\partial f_k}{\partial y_i}(\boldsymbol{g}(\boldsymbol{a})) \frac{\partial g_i}{\partial x_j}(\boldsymbol{a})
$$
\end{framei}


\begin{framev}[fs=small]{Helpful Calculation Rules}
Let $\boldsymbol{a}$, $\boldsymbol{b}$ denote vectors, $\boldsymbol{X}$, $\boldsymbol{A}$ matrices, and $f(\boldsymbol{X})^{-1}$ the inverse of $f(\boldsymbol{X})$ if it exists.
\begin{itemize}
\item $ \frac{d \boldsymbol{x}^{T} \boldsymbol{a}}{d \boldsymbol{x}}=\boldsymbol{a}^{T}$, $\frac{d \boldsymbol{a}^{T} \boldsymbol{x}}{d \boldsymbol{x}}=\boldsymbol{a}^{T}$
\item $\frac{d\boldsymbol{X}\boldsymbol{a}}{d\boldsymbol{a}} = \boldsymbol{X}$, $\frac{d\boldsymbol{a}^T\boldsymbol{X}}{d\boldsymbol{a}} = \boldsymbol{X}^T$
\item $\frac{d \boldsymbol{a}^{T} \boldsymbol{X} \boldsymbol{b}}{d \boldsymbol{X}}=\boldsymbol{a} \boldsymbol{b}^{T}$
\item $\frac{d \boldsymbol{x}^{T} \boldsymbol{A} \boldsymbol{x}}{d \boldsymbol{x}}=\boldsymbol{x}^{T}\left(\boldsymbol{A}+\boldsymbol{A}^{T}\right)$
\item For a symmetric matrix $\boldsymbol{W}$, $\frac{d}{d \boldsymbol{s}}(\boldsymbol{x}-\boldsymbol{A} \boldsymbol{s})^{T} \boldsymbol{W}(\boldsymbol{x}-\boldsymbol{A} \boldsymbol{s})=-2(\boldsymbol{x}-\boldsymbol{A} \boldsymbol{s})^{T} \boldsymbol{W} \boldsymbol{A}$
\item $\frac{d}{d \boldsymbol{X}} \boldsymbol{f}(\boldsymbol{X})^{T}=\left(\frac{d \boldsymbol{f}(\boldsymbol{X})}{d \boldsymbol{X}}\right)^{T}$
\item $\frac{d}{d \boldsymbol{X}} \boldsymbol{f}(\boldsymbol{X})^{-1}=-\boldsymbol{f}(\boldsymbol{X})^{-1} \frac{d \boldsymbol{f}(\boldsymbol{X})}{d \boldsymbol{X}} \boldsymbol{f}(\boldsymbol{X})^{-1}$
\item $\frac{d \boldsymbol{a}^{T} \boldsymbol{X}^{-1} \boldsymbol{b}}{d \boldsymbol{X}}=-\left(\boldsymbol{X}^{-1}\right)^{T} \boldsymbol{a} \boldsymbol{b}^{T}\left(\boldsymbol{X}^{-1}\right)^{T}$
\end{itemize}
\spacer
\begin{small}
\textbf{Note:} to compute gradients of matrices
with respect to vectors (or other matrices) we need \emph{tensors}, see chapter 5.4 of Deisenroth for more.
\end{small}
\end{framev}


\begin{framev}{Example: Logistic Regression I}
\begin{itemize}    \setlength{\itemsep}{0.5\baselineskip}
\item Let's say, for data in $\R^{n\times m}$ we're trying to minimize the risk in logistic regression by finding the gradient for negative log loss:
$$
-\ell({\theta})=\sum_{i=1}^n-y^{(i)} \log \left(\pi\left(\xv^{(i)} \mid \theta\right)\right)-\left(1-y^{(i)}\right) \log \left(1-\pi\left(\xv^{(i)} \mid \theta\right)\right)
$$
where $
\pi(\xv \mid {\theta})=s(f(\theta,\xv))
$ with
$f(\theta,\xv)=\theta^{T} \xv$ and $s(x)=\frac{1}{1+\exp (-x)}$.
\item[$\Rightarrow$] We want to find \begin{align*}
\nabla_\theta -\ell (\theta)=-\nabla_\theta \ell (\theta)&=-\sumin\underbrace{ y^{(i)} \log (s(f(\theta,\xv^{(i)})))+(1-y^{(i)}) \log (1-s(f(\theta,\xv^{(i)})))}_{=:y_i \log (s_i)+(1-y_i) \log (1-s_i)}
\end{align*}
\item $s_i = s(f_i)$; $h_i:=-[y_i \log (s_i)+(1-y_i) \log (1-s_i)]$; $f_i:=\theta^{T} \xv^{(i)}$
\end{itemize}
\end{framev}


\begin{framev}[fs=small]{Example: Logistic Regression II}
\begin{itemize}    \setlength{\itemsep}{0.5\baselineskip}
\item We can now directly apply the chain rule:
$$
-\nabla_\theta \ell (\theta)=\sumin\nabla_\theta h_i\circ s_i \circ f_i=\sumin\frac{d h_i}{d s_i} \frac{d s_i}{d f_i} \frac{d f_i}{d \theta}
$$
\item Given that, $\forall i\in\{1,\dots,n\}$ $$\frac{d h_i}{d s_i}=\frac{1-y_i}{1-s_i}-\frac{y_i}{s_i};\quad\frac{d s_i}{d f_i}=s(f_i)\big(1-s_i(f_i)\big);\quad\frac{d f_i}{d \theta}=\left(\xv^{(i)}\right)^T$$
\item we get that $-\nabla_\theta \ell (\theta)=\sumin\frac{d h_i}{d s_i} \frac{d s_i}{d f_i} \frac{d f_i}{d \theta}$ equals\begin{align*}
&\sumin \left[\frac{1-y^{(i)}}{1-s(f(\theta, \xv^{(i)}))}-\frac{y^{(i)}}{s(f(\theta, \xv^{(i)}))}\right] \cdot \left[s(f(\theta, \xv^{(i)}))(1-s(f(\theta, \xv^{(i)})))\right] \left(\xv^{(i)}\right)^T\\
=&\sumin\left(s\left(f\left(\theta,\xv^{(i)}\right)\right)-y^{(i)}\right)\left(\xv^{(i)}\right)^T\in\R^{1\times m}
\end{align*}
\end{itemize}
\end{framev}

\begin{framev}{Ex: Log. Regr. -- Alternative Notation}
This example highlights how using Leibniz notation can make things easier.\\
Alternatively, we could write this example out as follows:
\begin{itemize}
\item Given the sum rule, it suffices to find the derivative of
$$
-[y \log (s(f(\theta,\xv)))+(1-y) \log (1-s(f(\theta,\xv)))]
$$
\item Define $h(y,z):=-[y \log (z)+(1-y) \log (1-z)]$.
\item Now, what we are looking for is
$$
\nabla_\theta h(y,\cdot)\circ s \circ f(\cdot,\xv).
$$
where $f$ is a function from $\R^p$ to $\R$ and both $h$ and $s$ are functions from $\R$ to $\R$
$\Longrightarrow$ by the chain rule
$$
\nabla_\theta h(y,\cdot)\circ s \circ f(\cdot,\xv)=\frac{d h}{d s} \frac{d s}{d f} \frac{d f}{d \theta}\in\R^{1\times p}.
$$
\end{itemize}
\end{framev}


\begin{framev}{Ex: Log. Regr. -- Alternative Notation}
\begin{itemize}
\item Using the fact that $$\frac{d h}{d z}=\frac{1-y}{1-z}-\frac{y}{z};\quad\frac{d s}{d x}=s(x)\big(1-s(x)\big);\quad\frac{d f}{d \theta}=\xv$$ we get \begin{align*}
[\nabla_\theta h(y,\cdot)\circ s \circ f(\cdot,\xv)]_j&=\left[\frac{1-y}{1-s(f(\theta,\xv))}-\frac{y}{s(f(\theta,\xv))}\right] \cdot \left[s(f(\theta,\xv))(1-s(f(\theta,\xv)))\right]\cdot \frac{\partial f}{\partial\theta_j}\\
&=(s(f(\theta,\xv))-y)\xv_j
\end{align*}
$\Longrightarrow \nabla_\theta h(y,\cdot)\circ s \circ f(\cdot,\xv)=(s(f(\theta,\xv))-y)\xv^T$
\item Plugging this back into the original sum term we get $$
-\nabla_\theta \ell (\theta)=\sumin\left(s\left(f\left(\theta,\xv^{(i)}\right)\right)-y^{(i)}\right)\left(\xv^{(i)}\right)^T\in\R^{1\times m},
$$ the same as before.
\end{itemize}
\end{framev}
