\begin{framei}[sep=0.5em]{Finding Pareto-optimal solutions I}
\item \textbf{Weighted sum scalarization} solves:
$$\xvs := \argmin_{\xv \in \S } \sumim w_i f_i(\xv)$$
We can show that this approach can find Pareto-optimal solutions.
\item Without loss of generality we can assume $\sumid w_i =1$.
\item \textbf{Theorem:} If $w_i>0$ for all $i$, then $\xvs$ is Pareto-optimal.
\spacer
\textbf{Proof.} Suppose $\xvs$ is not Pareto-optimal, i.e., there exists $\xv$ such that $f_i(\xv) \leq f_i(\xvs)$ for all $i$ and $f_j(\xv) < f_j(\xvs)$ for at least one $j$. Then,
$$\sumim w_i f_i(\xv) < \sumid w_i f_i(\xvs),$$
which contradicts the definition of $\xvs$ as the minimizer.
\end{framei}

\begin{framev}{Finding Pareto-optimal solutions V}
\textbf{Theorem (Geoffrion, 1968):} Let $\S$ be \underline{convex} and assume $f_1,\dots,f_d$ are \underline{convex}. Then $\xvs \in \S$ is \underline{properly} Pareto-optimal if and only if
$\xvs$ is an optimal solution of
$$\min_{\xv \in \S } \sumim w_i f_i(\xv)$$
with strictly positive weights $w_i$ for all $i$.
\begin{itemize}
\item For convex functions and convex search spaces, we can find all \underline{properly} Pareto-optimal solutions using weighted sum scalarization.
\item \textbf{Proper Pareto-optimal solution}: Exclude solutions where an arbitrary small improvement in one objective leads to an arbitrary large deterioration in another objective.
\end{itemize}
\end{framev}


\begin{framev}{Finding Pareto-optimal solutions VI}
\textbf{Definition (Geoffrion, 1968):} A Pareto-optimal solution $\xvs \in \S$ is called \textbf{properly} Pareto-optimal, if there exists a constant $M>0$ such that for all $i$ and ${\xv} \in \S$ with $f_i(\xv) < f_i(\xvs)$ there exists an index $j$, such that $f_j(\xvs) < f_j(\xv)$ and
$$ \frac{f_i(\xvs)-f_i(\xv)}{f_j(\xv)-f_j(\xvs)} \leq M$$
\textbf{Example:} $f_1(x) =(x-1)^2$ and $f_2(x) = 3(x-2)^2$ for $0 \le x \le 3$.
\begin{itemize}
\item Pareto-optimal solutions are $x \in [1,2]$ (as exercise).
\item $\xs=1$ is not properly Pareto-optimal.\\
\textbf{Proof.} For $\xs=1$ we have $f_1(\xs)=0$ and $f_2(\xs)=3$. Let $x=1+\eps$ for $\eps>0$, then $f_2(x)<f_2(\xs)$ and $f_1(x)>f_1(\xs)$. Then,
$$\frac{f_2(\xs)-f_2(x)}{f_1(x)-f_1(\xs)} = \frac{3- 3(\eps-1)^2}{\eps^2} =\frac{6\eps-3\eps^2}{\eps^2}\to \infty \text{ for } \eps \to 0$$
\end{itemize}
\end{framev}

\begin{framei}{Finding Pareto-optimal solutions IX}
\item General form of the $\eps$-constraint method:
$$\xvs := \argmin_{\xv \in \S} f_i(\xv) \quad \text{s.t. } f_j(\xv) \leq \eps_j, \quad j=1,\dots,m, j \neq i.$$
\item Are solutions found by the $\eps$-constraint method Pareto-optimal?
\item \textbf{Theorem:} If $\xvs$ is the \underline{unique} optimal solution of the $\eps$-constraint method \underline{for some} index $i$, then $\xvs$ is Pareto-optimal.\\
\spacer
\textbf{Proof.} Suppose $\xvs$ is not Pareto-optimal. Then there exists $\xv \in \S$, such that $f_i(\xv) \leq f_i(\xvs)$ for all $i = 1, \dots, m$ and $f_j(\xv) < f_j(\xvs)$ for at least one $j \in \{1, \dots, m\}$.
Since $\xvs$ is feasible for the $\eps$-constraint problem, it follows that $f_i(\xv) \leq f_i(\xvs) \leq \eps_i$, so $\xv$ is also feasible.
Moreover, $f_1(\xv) \leq f_1(\xvs)$, and since $\xvs$ is the unique minimizer of $\text{cost}_1$, we must have $\xv = \xvs$, which is a contradiction.
\end{framei}


\begin{framei}{Finding Pareto-optimal solutions X}
\item Can the $\eps$-constraint method find all Pareto-optimal solutions?
$$\xvs := \argmin_{\xv \in \S} f_i(\xv) \quad \text{s.t. } f_j(\xv) \leq \eps_j, \quad j=1,\dots,m, j \neq i.$$
\item \textbf{Theorem:} $\xvs \in \S$ is Pareto-optimal if and only if there exists $\mathbf{\eps} \in \R^m$ such that $\xvs$ is an optimal solution \underline{for all} $i=1,\dots,m$.\\
\spacer
\textbf{Proof.} ``$\Rightarrow$'' Suppose $\xvs$ is not an optimal solution of the $\eps$-constraint problem. Then define $\mathbf{\eps}=f(\xvs)$, and there exists $\xv \in \S$ with $f_i(\xv) <f_i(\xvs)$ and $f_j(\xv) \leq \eps_j = f_j(\xvs)$, which contradicts the Pareto-optimality of $\xvs$.\\
``$\Leftarrow$'' Suppose $\xvs$ is not Pareto-optimal. Then there exists $\xv \in \S$, such that $f_i(\xv) < f_i(\xvs)$ for at least one $i \in \{1, \dots, m\}$ and $f_j(\xv) \leq f_j(\xvs)$ for all $j = 1, \dots, m$.
Since $\xvs$ is a solution of the $\eps$-constraint problem, it follows that $f_j(\xv) \leq f_j(\xvs) \leq \eps_j$, so $\xv$ satisfies the constraints, which contradicts the optimality of $\xvs$ for the $\eps$-constraint problem and index $i$.
\end{framei}


\begin{framev}[fs=footnotesize]{Evaluation of solutions}
\splitV[0.5]{
A common measure for the performance of a set of candidates
$\mathcal{P}\subset\S$ is the \textbf{dominated hypervolume}:
$$
S(\mathcal{P},R)
= \lambda\Bigl(
\bigcup_{\tilde{\xv}\in \mathcal{P}}
\{\;\xv \;\mid\;
\tilde{\xv}\prec \xv \prec R
\}
\Bigr),
$$
where $\lambda(\cdot)$ is the Lebesgue measure and $R$ a pre-specified reference point that every point in the set must dominate.
}{
\imageC[0.6]{figure_man/dominated_hypervolume.png}
}
\begin{itemize}
\item HV is measured w.r.t.\ the reference point $R$, often chosen so each
component is a known ``worst'' (upper bound) for that objective.
\item HV is also called the \textbf{S-metric}.
\item Its computation is $\mathcal{O}(n^{m-1})$ in the number of points
and dimension $m$.
\item Efficient algorithms exist for small $m$.
In ML, we rarely exceed $m=3$ in practice.
\end{itemize}
\end{framev}
